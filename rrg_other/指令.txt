hf mf chk --dump -k FFFFFFFFFFFF -k 000000000000 -k A0A1A2A3A4A5 -k B0B1B2B3B4B5

hf mf dump -f ../keys/ID_xxx -k hf-mf-FA9B36C5-key.bin --1k
